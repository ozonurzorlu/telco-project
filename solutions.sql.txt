

-- 1.1 'Kobiye Destek' tarifesine abone olan müşterileri listeleyin.
SELECT c.CUSTOMER_ID, c.NAME, c.CITY, t.NAME as TARIFF_NAME
FROM CUSTOMERS c
JOIN TARIFFS t ON c.TARIFF_ID = t.TARIFF_ID
WHERE t.NAME = 'Kobiye Destek';

-- 1.2 Bu tarifeye abone olan en yeni müşteriyi bulun.
SELECT * FROM (
    SELECT c.CUSTOMER_ID, c.NAME, c.SIGNUP_DATE
    FROM CUSTOMERS c
    JOIN TARIFFS t ON c.TARIFF_ID = t.TARIFF_ID
    WHERE t.NAME = 'Kobiye Destek'
    ORDER BY c.SIGNUP_DATE DESC
)
WHERE ROWNUM = 1;

-- 2.1 Tarife bazlı müşteri dağılımı (Hangi tarifede kaç kişi var?)
SELECT t.NAME AS TARIFF_NAME, COUNT(c.CUSTOMER_ID) AS TOTAL_CUSTOMERS
FROM TARIFFS t
LEFT JOIN CUSTOMERS c ON t.TARIFF_ID = c.TARIFF_ID
GROUP BY t.NAME
ORDER BY TOTAL_CUSTOMERS DESC;

-- 3.1 En çok müşterisi olan ilk 3 şehir
SELECT * FROM (
    SELECT CITY, COUNT(CUSTOMER_ID) AS CUSTOMER_COUNT
    FROM CUSTOMERS
    GROUP BY CITY
    ORDER BY CUSTOMER_COUNT DESC
)
WHERE ROWNUM <= 3;

-- 4.1 İnternet (data) kullanımı 0 olan müşteriler
SELECT c.NAME, c.CITY, s.DATA_USAGE
FROM CUSTOMERS c
JOIN MONTHLY_STATS s ON c.CUSTOMER_ID = s.CUSTOMER_ID
WHERE s.DATA_USAGE = 0;

-- 5.1 Şehirlerin ortalama internet kullanım miktarları
SELECT c.CITY, ROUND(AVG(s.DATA_USAGE), 2) AS AVG_DATA_USAGE
FROM CUSTOMERS c
JOIN MONTHLY_STATS s ON c.CUSTOMER_ID = s.CUSTOMER_ID
GROUP BY c.CITY
ORDER BY AVG_DATA_USAGE DESC;

-- 6.1 Her bir tarifenin şirkete getirdiği toplam gelir
SELECT t.NAME AS TARIFF_NAME, 
       COUNT(c.CUSTOMER_ID) * t.MONTHLY_FEE AS TOTAL_REVENUE
FROM TARIFFS t
JOIN CUSTOMERS c ON t.TARIFF_ID = c.TARIFF_ID
GROUP BY t.NAME, t.MONTHLY_FEE
ORDER BY TOTAL_REVENUE DESC;
